#include "stdafx.h"
#include <iostream>
#include <string>

using namespace std;

int main(void)
{
	int t;
	while (true)
	{
		cin.setf(ios_base::hex, ios_base::basefield);
		cin >> t;
		if (cin.fail())
		{
			cout << "제대로 된 값을 입력해주세요." << endl;
			cin.clear();
			cin.ignore(100, '\n');
		}
		else cout << "입력 : " << t << endl;

		if (t == 0) break;
	}
}
