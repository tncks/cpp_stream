	string name;
	int age;

	Human(const string& name, int age) : name(name), age(age) {
	}
	string get_info()
	{
		return "Name::" + name + " / Age::" + to_string(age);
	}

	friend ofstream& operator << (ofstream& o, Human& h);
}

ofstream& operator << (ofstream& o, Human& h)
{
	o << h.get_info();
	return o;
